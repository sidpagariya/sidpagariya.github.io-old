19   def countbits(n): 
20       'Returns the count of bits set to 1 in a positive integer' 
21       k = 0 
22       if n == 0: 
23           return '0' 
24       numToConvert = n 
25       while numToConvert > 0: 
26           lastDigit = numToConvert % 2 
27           if lastDigit == 1: 
28               k+=1 
29           numToConvert //= 2 
30       return k

39   # Returns a string that represents a given non-negative

41   # Precondition: n is an int; n >= 0 
42   def intToBin(n): 
43       '''Converts n >= 0 into a string of binary digits 
44       and returns that string''' 
45       s = "" 
46       if n == 0: 
47           return '0' 
48       numToConvert = n 
49       while numToConvert > 0: 
50           lastDigit = numToConvert % 2 
51           s = str(lastDigit) + s 
52           numToConvert //= 2 
53       return s
