21   def intToHex(n): 
22       s = '' 
23       hdigs = '0123456789abcdef' 
24       count = 1 
25       while count <= 8: 
26           l4bits = (n & 0x0f) 
27           s = hdigs[l4bits] + s 
28           n >>= 4 
29           count += 1 
30       return s

39   def revBits(n): 
40       revv=0 
41       abit = 0x01 
42       rBit = 0x20000000 
43       while rBit != 0: 
44           if (n & abit) != 0: 
45               revv |= rBit 
46           abit <<= 1 
47           rBit >>= 1 
48       return revv
